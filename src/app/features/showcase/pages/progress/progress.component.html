// Indeterminate progress bar

// Basic progress spinner

// Determinate spinner with custom size
<mat-progress-spinner 
  mode="determinate" 
  [value]="progress"
  diameter="50"
  strokeWidth="4">
</mat-progress-spinner>

// Different colors
<mat-progress-bar color="accent" mode="indeterminate"></mat-progress-bar>
