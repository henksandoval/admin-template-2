Code Example 
// Basic chip set
<mat-chip-set>
  <mat-chip>One fish</mat-chip>
  <mat-chip color="accent">Two fish</mat-chip>
</mat-chip-set>

// Removable chips
<mat-chip-set>
  <mat-chip *ngFor="let fruit of fruits" (removed)="removeChip(fruit)">
    {{fruit}}
    <button matChipRemove>
      <mat-icon>cancel</mat-icon>
    </button>
  </mat-chip>
</mat-chip-set>

// Selectable chips (single)
<mat-chip-listbox>
  <mat-chip-option>Small</mat-chip-option>
  <mat-chip-option selected>Medium</mat-chip-option>
  <mat-chip-option>Large</mat-chip-option>
</mat-chip-listbox>

// Multiple selection
<mat-chip-listbox multiple>
  <mat-chip-option selected>Bold</mat-chip-option>
  <mat-chip-option>Italic</mat-chip-option>
</mat-chip-listbox>

// Chips with icons
<mat-chip>
  <mat-icon matChipAvatar>home</mat-icon>
  Home
</mat-chip>
