Code Example 
// Simple list
<mat-list>
  <mat-list-item>Item 1</mat-list-item>
  <mat-list-item>Item 2</mat-list-item>
</mat-list>

// List with icons
<mat-list>
  <mat-list-item>
    <mat-icon matListItemIcon>home</mat-icon>
    Home
  </mat-list-item>
</mat-list>

// Multi-line list with avatar
<mat-list>
  <mat-list-item>
    <img matListItemAvatar src="avatar.jpg" alt="Avatar">
    <span matListItemTitle>John Doe</span>
    <span matListItemLine>john.doe@example.com</span>
    <span matListItemLine>+1 234 567 8900</span>
  </mat-list-item>
</mat-list>

// Navigation list
<mat-nav-list>

    <mat-icon matListItemIcon>dashboard</mat-icon>
    <span matListItemTitle>Dashboard</span>
  </mat-list-item>
