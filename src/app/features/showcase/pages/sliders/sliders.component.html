Code Example 
// Basic slider
<mat-slider min="0" max="100" step="1">
  <input matSliderThumb [(ngModel)]="value">
</mat-slider>

// Discrete slider with tick marks
<mat-slider min="0" max="10" step="1" discrete showTickMarks>
  <input matSliderThumb [value]="5">
</mat-slider>

// Range slider
<mat-slider min="0" max="1000" step="10">
  <input matSliderStartThumb [value]="100">
  <input matSliderEndThumb [value]="400">
</mat-slider>

// Vertical slider
<mat-slider min="0" max="100" vertical>
  <input matSliderThumb [value]="70">
</mat-slider>

// Different colors
<mat-slider color="accent" min="0" max="100">
