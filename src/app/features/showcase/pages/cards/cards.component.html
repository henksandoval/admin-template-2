Code Examples 
// Basic Card
<mat-card>
  <mat-card-header>
    <mat-card-title>Card Title</mat-card-title>
    <mat-card-subtitle>Card subtitle</mat-card-subtitle>
  </mat-card-header>
  <mat-card-content>
    Card content goes here
  </mat-card-content>
  <mat-card-actions>
    <button mat-button>ACTION</button>
  </mat-card-actions>
</mat-card>

// Card with Image
<mat-card>
  <img mat-card-image src="image.jpg" alt="Description">
  <mat-card-header>
    <mat-card-title>Image Card</mat-card-title>
  </mat-card-header>
  <mat-card-content>
    Content below image
  </mat-card-content>
</mat-card>

// Card with Avatar
<mat-card>
  <mat-card-header>
    <div mat-card-avatar>A</div>
    <mat-card-title>User Name</mat-card-title>
    <mat-card-subtitle>User Role</mat-card-subtitle>
  </mat-card-header>
</mat-card>
