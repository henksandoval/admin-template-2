Code Example 
// Basic tab group
<mat-tab-group>
  <mat-tab label="First">
    <div class="p-4">
      <h3>First Tab Content</h3>
      <p>Content goes here</p>
    </div>
  </mat-tab>
  <mat-tab label="Second">
    <div class="p-4">Second tab content</div>
  </mat-tab>
</mat-tab-group>

// Tab with icon
<mat-tab>
  <ng-template mat-tab-label>
    <mat-icon>home</mat-icon>
    Home
  </ng-template>
  <div>Tab content</div>
</mat-tab>

// Colored tab group
<mat-tab-group backgroundColor="primary" color="accent">
  <mat-tab label="Tab 1">Content 1</mat-tab>
  <mat-tab label="Tab 2" [disabled]="true">Content 2</mat-tab>
</mat-tab-group>

// Dynamic tabs
<mat-tab-group [(selectedIndex)]="selectedTabIndex">
  <mat-tab *ngFor="let tab of dynamicTabs" [label]="tab.label">
    {{tab.content}}
  </mat-tab>
</mat-tab-group>
