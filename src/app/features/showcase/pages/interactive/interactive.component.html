Dialog Examples

Note: Dialog functionality would require creating separate dialog components. This is a simplified example showing the trigger buttons.

Dialog Implementation Notes

Dialogs in Angular Material require creating separate components for the dialog content. You would use MatDialog.open() to display them. Each dialog can have custom content, actions, and can return data when closed.

Code Example 
// Basic tooltip
<button mat-raised-button 
        matTooltip="This is a tooltip"
        matTooltipPosition="above">
  Hover me
</button>

// Basic menu
<button mat-raised-button [matMenuTriggerFor]="menu">
  Open Menu
</button>
<mat-menu #menu="matMenu">
  <button mat-menu-item>
    <mat-icon>save</mat-icon>
    Save
  </button>
  <button mat-menu-item>Edit</button>
</mat-menu>

// Nested menu
<mat-menu #parentMenu="matMenu">
  <button mat-menu-item [matMenuTriggerFor]="childMenu">
    Submenu
  </button>
</mat-menu>
<mat-menu #childMenu="matMenu">
  <button mat-menu-item>Option 1</button>
</mat-menu>

// Dialog (in component)
constructor(private dialog: MatDialog) {}

openDialog() {
  this.dialog.open(MyDialogComponent, {
    width: '400px',
    data: { message: 'Hello!' }
  });
}
